    def run_evaluation(
        self,
        dataset: Dataset,
        metrics: list[str],
        thresholds: dict[str, float],
        *,
        parallel: bool = False,
        batch_size: int = 5,

        # ë¹„ë™ê¸° ì‹¤í–‰ with ì§„í–‰ ìƒí™© ì½œë°±

                metrics=metrics,
                llm=self._llm,
                thresholds=thresholds,
                parallel=parallel,
                batch_size=batch_size,
                progress_callback=progress_callback,
            )

        return asyncio.run(run_with_progress())

    def get_improvement_guide(
        self,
        run: EvaluationRun,
        *,
        include_llm: bool = False,
    ) -> ImprovementReport:
        """ê°œì„  ê°€ì´ë“œ ìƒì„±."""
        detector = PatternDetector()
        generator = InsightGenerator(llm_adapter=self._llm) if include_llm else None

        service = ImprovementGuideService(
            pattern_detector=detector,
            insight_generator=generator,
        )

        return service.generate_report(run, include_llm_analysis=include_llm)

    def check_quality_gate(
        self,
        run: EvaluationRun,
        thresholds: dict[str, float] | None = None,
    ) -> GateReport:
        """í’ˆì§ˆ ê²Œì´íŠ¸ ì²´í¬."""
        thresholds = thresholds or run.thresholds

        results = []
        for metric in run.metrics_evaluated:
            score = run.get_avg_score(metric)
            threshold = thresholds.get(metric, 0.7)

            results.append(GateResult(
                metric=metric,
                score=score,
                threshold=threshold,
                passed=score >= threshold,
                gap=threshold - score,
            ))

        return GateReport(
            run_id=run.run_id,
            results=results,
            overall_passed=all(r.passed for r in results),
        )
